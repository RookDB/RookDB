8/// Initialize a new table file
9pub fn init_table(file: &mut File) -> io::Result<()> {
10    // Move cursor to the beginning of the file
11    file.seek(SeekFrom::Start(0))?;
12
13    // Allocate 8192 (TABLE_HEADER_SIZE) + 8192 (PAGE_SIZE) bytes = 16KB
14    let mut zero_buf = vec![0u8; TABLE_HEADER_SIZE as usize];
15
16    //  Write "1" into the first 4 bytes (little-endian u32)
17    // This can represent the total number of pages, e.g. 1
18    zero_buf[0..4].copy_from_slice(&1u32.to_le_bytes());
19
20    // Write the full buffer (header) to the file
21    file.write_all(&zero_buf)?;
22
23    // Optionally, flush to ensure write is committed
24    file.flush()?;
25    file.sync_all()?;
26
27    create_page(file)?;
28
29    Ok(())
30}

33pub fn insert_tuple(file: &mut File, data: &[u8]) -> io::Result<()> {
34    let mut total_pages = page_count(file)?;
35    let mut last_page_num = total_pages - 1;
36
37    let mut page = Page::new();
38    read_page(file, &mut page, last_page_num)?;
39
40    let free_space = page_free_space(&page)?;
41    let required = data.len() as u32 + ITEM_ID_SIZE;
42
43    if required > free_space {
44        create_page(file)?;
45        total_pages += 1;
46        last_page_num = total_pages - 1;
47        read_page(file, &mut page, last_page_num)?;
48    }
49
50    let mut lower = u32::from_le_bytes(page.data[0..4].try_into().unwrap());
51    let mut upper = u32::from_le_bytes(page.data[4..8].try_into().unwrap());
52
53    let start = upper - data.len() as u32;
54    page.data[start as usize..upper as usize].copy_from_slice(data);
55
56    upper = start;
57    page.data[4..8].copy_from_slice(&upper.to_le_bytes());
58
59    page.data[lower as usize..lower as usize + 4].copy_from_slice(&start.to_le_bytes());
60    page.data[lower as usize + 4..lower as usize + 8]
61        .copy_from_slice(&(data.len() as u32).to_le_bytes());
62
63    lower += ITEM_ID_SIZE;
64    page.data[0..4].copy_from_slice(&lower.to_le_bytes());
65
66    write_page(file, &mut page, last_page_num)?;
67    Ok(())
68}
