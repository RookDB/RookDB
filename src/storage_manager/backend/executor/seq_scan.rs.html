1use std::fs::File;

3
4use crate::catalog::types::Catalog;

6use crate::page::{Page, PAGE_HEADER_SIZE, ITEM_ID_SIZE};
7use crate::table::page_count;
8
9pub fn show_tuples(
10    catalog: &Catalog,
11    db_name: &str,
12    table_name: &str,
13    file: &mut File,
14) -> io::Result<()> {
15    // 1. Get schema from catalog
16    let db = catalog
17        .databases
18        .get(db_name)
19        .ok_or_else(|| io::Error::new(io::ErrorKind::NotFound, format!("Database '{}' not found", db_name)))?;
20
21    let table = db
22        .tables
23        .get(table_name)
24        .ok_or_else(|| io::Error::new(io::ErrorKind::NotFound, format!("Table '{}' not found", table_name)))?;
25
26    let columns = &table.columns;
27
28    // 2. Read total number of pages
29    file.seek(SeekFrom::Start(0))?;
30    let mut buf = [0u8; 4];
31    file.read_exact(&mut buf)?;
32    let mut total_pages = u32::from_le_bytes(buf);
33
34    println!("\n=== Tuples in '{}.{}' ===", db_name, table_name);
35    println!("Total pages: {}", total_pages);
36    total_pages = total_pages;
37
38    // 3. Loop through each page
39    for page_num in 1..total_pages {
40        let mut page = Page::new();
41        read_page(file, &mut page, page_num)?;
42        println!("\n-- Page {} --", page_num);
43
44        let lower = u32::from_le_bytes(page.data[0..4].try_into().unwrap());
45        let upper = u32::from_le_bytes(page.data[4..8].try_into().unwrap());
46        println!("Lower: {}, Upper: {}", lower, upper);
47        let num_items = (lower - PAGE_HEADER_SIZE) / ITEM_ID_SIZE;
48
49        println!("Lower: {}, Upper: {}, Tuples: {}", lower, upper, num_items);
50
51        // 4. For each tuple
52        for i in 0..num_items {
53            let base = (PAGE_HEADER_SIZE + i * ITEM_ID_SIZE) as usize;
54            let offset = u32::from_le_bytes(page.data[base..base + 4].try_into().unwrap());
55            let length = u32::from_le_bytes(page.data[base + 4..base + 8].try_into().unwrap());
56            let tuple_data = &page.data[offset as usize..(offset + length) as usize];
57
58            print!("Tuple {}: ", i + 1);
59
60            // 5. Decode each column
61            let mut cursor = 0usize;
62            for col in columns {
63                match col.data_type.as_str() {
64                    "INT" => {
65                        if cursor + 4 <= tuple_data.len() {
66                            let val = i32::from_le_bytes(tuple_data[cursor..cursor + 4].try_into().unwrap());
67                            print!("{}={} ", col.name, val);
68                            cursor += 4;
69                        }
70                    }
71                    "TEXT" => {
72                        if cursor + 10 <= tuple_data.len() {
73                            let text_bytes = &tuple_data[cursor..cursor + 10];
74                            let text = String::from_utf8_lossy(text_bytes).trim().to_string();
75                            print!("{}='{}' ", col.name, text);
76                            cursor += 10;
77                        }
78                    }
79                    _ => {
80                        print!("{}=<unsupported> ", col.name);
81                    }
82                }
83            }
84            println!();
85        }
86    }
87
88    println!("\n=== End of tuples ===\n");
89    Ok(())
90}
