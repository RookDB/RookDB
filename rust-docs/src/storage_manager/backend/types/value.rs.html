15/// IEEE 754 `f32` wrapper that satisfies `Eq` by comparing bit patterns.
16/// NaN values with identical bit patterns compare equal.
17#[derive(Debug, Clone, Copy, Serialize, Deserialize)]
18#[serde(transparent)]
19pub struct OrderedF32(pub f32);
20
21impl PartialEq for OrderedF32 {
22    fn eq(&self, other: &Self) -> bool {
23        self.0.to_bits() == other.0.to_bits()
24    }
25}
26impl Eq for OrderedF32 {}
27
28impl PartialOrd for OrderedF32 {
29    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
30        self.0.partial_cmp(&other.0)
31    }
32}
33
34impl Ord for OrderedF32 {
35    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
36        self.partial_cmp(other)
37            .unwrap_or_else(|| match (self.0.is_nan(), other.0.is_nan()) {
38                (true, true) => std::cmp::Ordering::Equal,
39                (true, false) => std::cmp::Ordering::Greater,
40                _ => std::cmp::Ordering::Less,
41            })
42    }
43}

91impl fmt::Display for DataValue {
92    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
93        match self {
94            DataValue::SmallInt(v) => write!(f, "{}", v),
95            DataValue::Int(v) => write!(f, "{}", v),
96            DataValue::BigInt(v) => write!(f, "{}", v),
97            DataValue::Real(v) => write!(f, "{}", v.0),
98            DataValue::DoublePrecision(v) => write!(f, "{}", v.0),
99            DataValue::Numeric(v) => {
100                let sign = if v.unscaled < 0 { "-" } else { "" };
101                let mut digits = v.unscaled.abs().to_string();
102                let scale = v.scale as usize;
103                if scale == 0 {
104                    write!(f, "{}{}", sign, digits)
105                } else {
106                    if digits.len() <= scale {
107                        digits = format!("{}{}", "0".repeat(scale + 1 - digits.len()), digits);
108                    }
109                    let split = digits.len() - scale;
110                    write!(f, "{}{}.{}", sign, &digits[..split], &digits[split..])
111                }
112            }
113            DataValue::Bool(v) => write!(f, "{}", v),
114            DataValue::Char(v) => write!(f, "'{}'", v.trim_end_matches(' ')),
115            DataValue::Varchar(v) => write!(f, "'{}'", v),
116            DataValue::Date(v) => write!(f, "{}", v.format("%Y-%m-%d")),
117            DataValue::Time(v) => write!(f, "{}", v.format("%H:%M:%S%.6f")),
118            DataValue::Bit(v) => write!(f, "B'{}'", v),
119            DataValue::Timestamp(v) => write!(f, "{}", v.format("%Y-%m-%d %H:%M:%S%.6f")),
120        }
121    }
122}

124fn parse_numeric_literal(input: &str, precision: u8, scale: u8) -> Result<NumericValue, String> {
125    let raw = input.trim().trim_matches('"').trim_matches('\'');
126    if raw.is_empty() {
127        return Err("NUMERIC value cannot be empty".to_string());
128    }
129
130    let (sign, body) = if let Some(rest) = raw.strip_prefix('-') {
131        (-1_i128, rest)
132    } else if let Some(rest) = raw.strip_prefix('+') {
133        (1_i128, rest)
134    } else {
135        (1_i128, raw)
136    };
137
138    let mut parts = body.split('.');
139    let int_part = parts.next().unwrap_or("");
140    let frac_part = parts.next();
141    if parts.next().is_some() {
142        return Err(format!(
143            "Invalid NUMERIC value '{}': too many decimal points",
144            raw
145        ));
146    }
147    if !int_part.chars().all(|c| c.is_ascii_digit()) {
148        return Err(format!(
149            "Invalid NUMERIC value '{}': invalid integer digits",
150            raw
151        ));
152    }
153    let frac = frac_part.unwrap_or("");
154    if !frac.chars().all(|c| c.is_ascii_digit()) {
155        return Err(format!(
156            "Invalid NUMERIC value '{}': invalid fractional digits",
157            raw
158        ));
159    }
160    if frac.len() > scale as usize {
161        return Err(format!(
162            "NUMERIC({}, {}) value '{}' has too many fractional digits",
163            precision, scale, raw
164        ));
165    }
166
167    let mut combined = String::new();
168    if int_part.is_empty() {
169        combined.push('0');
170    } else {
171        combined.push_str(int_part);
172    }
173    combined.push_str(frac);
174    combined.push_str(&"0".repeat(scale as usize - frac.len()));
175
176    let normalized = combined.trim_start_matches('0');
177    let effective_digits = if normalized.is_empty() {
178        1
179    } else {
180        normalized.len()
181    };
182    if effective_digits > precision as usize {
183        return Err(format!(
184            "NUMERIC({}, {}) value '{}' exceeds precision {}",
185            precision, scale, raw, precision
186        ));
187    }
188
189    let unscaled_abs = combined
190        .parse::<i128>()
191        .map_err(|_| format!("NUMERIC value '{}' is out of supported range", raw))?;
192    Ok(NumericValue {
193        unscaled: sign * unscaled_abs,
194        scale,
195    })
196}

198fn encode_numeric_bcd(value: &NumericValue, precision: u8) -> Result<Vec<u8>, String> {
199    let mut digits = value.unscaled.abs().to_string();
200    if digits.len() > precision as usize {
201        return Err(format!(
202            "NUMERIC value '{}' exceeds precision {}",
203            value.unscaled, precision
204        ));
205    }
206    if digits.len() < precision as usize {
207        digits = format!(
208            "{}{}",
209            "0".repeat(precision as usize - digits.len()),
210            digits
211        );
212    }
213
214    let mut nibbles: Vec<u8> = Vec::with_capacity(precision as usize + 2);
215    if precision % 2 == 0 {
216        nibbles.push(0);
217    }
218    for ch in digits.chars() {
219        nibbles.push((ch as u8) - b'0');
220    }
221    nibbles.push(if value.unscaled < 0 { 0x0D } else { 0x0C });
222
223    let mut out = Vec::with_capacity(nibbles.len().div_ceil(2));
224    for pair in nibbles.chunks(2) {
225        let hi = pair[0] & 0x0F;
226        let lo = pair.get(1).copied().unwrap_or(0) & 0x0F;
227        out.push((hi << 4) | lo);
228    }
229    Ok(out)
230}

232fn decode_numeric_bcd(bytes: &[u8], precision: u8, scale: u8) -> Result<NumericValue, String> {
233    let expected = ((precision as usize) + 1).div_ceil(2);
234    if bytes.len() < expected {
235        return Err(format!(
236            "NUMERIC({}, {}) requires {} bytes",
237            precision, scale, expected
238        ));
239    }
240
241    let mut nibbles = Vec::with_capacity(expected * 2);
242    for b in &bytes[..expected] {
243        nibbles.push((b >> 4) & 0x0F);
244        nibbles.push(b & 0x0F);
245    }
246
247    let start = if precision % 2 == 0 { 1 } else { 0 };
248    let digits_slice = &nibbles[start..start + precision as usize];
249    let sign_nibble = nibbles[start + precision as usize];
250    if !digits_slice.iter().all(|d| *d <= 9) {
251        return Err("NUMERIC payload contains invalid BCD digit".to_string());
252    }
253    let mut digits = String::with_capacity(precision as usize);
254    for d in digits_slice {
255        digits.push((b'0' + *d) as char);
256    }
257    let abs_val = digits
258        .parse::<i128>()
259        .map_err(|_| "NUMERIC decoded value is out of range".to_string())?;
260    let sign = match sign_nibble {
261        0x0C => 1_i128,
262        0x0D => -1_i128,
263        _ => return Err("NUMERIC payload has invalid sign nibble".to_string()),
264    };
265    Ok(NumericValue {
266        unscaled: sign * abs_val,
267        scale,
268    })
269}

272    pub fn to_bytes(&self) -> Vec<u8> {
273        match self {
274            DataValue::SmallInt(v) => v.to_le_bytes().to_vec(),
275            DataValue::Int(v) => v.to_le_bytes().to_vec(),
276            DataValue::BigInt(v) => v.to_le_bytes().to_vec(),
277            DataValue::Real(v) => v.0.to_le_bytes().to_vec(),
278            DataValue::DoublePrecision(v) => v.0.to_le_bytes().to_vec(),
279            DataValue::Numeric(v) => {
280                // This variant needs precision from DataType; callers should prefer to_bytes_for_type.
281                let mut out = Vec::with_capacity(18);
282                out.extend_from_slice(&v.unscaled.to_le_bytes());
283                out.push(v.scale);
284                out
285            }
286            DataValue::Bool(v) => vec![u8::from(*v)],
287            DataValue::Char(v) => v.as_bytes().to_vec(),
288            DataValue::Varchar(v) => {
289                let bytes = v.as_bytes();
290                let mut out = Vec::with_capacity(2 + bytes.len());
291                out.extend_from_slice(&(bytes.len() as u16).to_le_bytes());
292                out.extend_from_slice(bytes);
293                out
294            }
295            DataValue::Date(v) => {
296                let epoch = NaiveDate::from_ymd_opt(1970, 1, 1).expect("valid epoch");
297                let days = v.signed_duration_since(epoch).num_days() as i32;
298                days.to_le_bytes().to_vec()
299            }
300            DataValue::Time(v) => {
301                // microseconds since midnight
302                let micros = v.num_seconds_from_midnight() as i64 * 1_000_000
303                    + v.nanosecond() as i64 / 1_000;
304                micros.to_le_bytes().to_vec()
305            }
306            DataValue::Bit(v) => pack_bit_string(v),
307            DataValue::Timestamp(v) => {
308                // microseconds since Unix epoch
309                let epoch = NaiveDate::from_ymd_opt(1970, 1, 1)
310                    .unwrap()
311                    .and_hms_opt(0, 0, 0)
312                    .unwrap();
313                let micros = v
314                    .signed_duration_since(epoch)
315                    .num_microseconds()
316                    .unwrap_or(i64::MAX);
317                micros.to_le_bytes().to_vec()
318            }
319        }
320    }

322    pub fn from_bytes(ty: &DataType, bytes: &[u8]) -> Result<Self, String> {
323        match ty {
324            DataType::SmallInt => {
325                if bytes.len() < 2 {
326                    return Err("SMALLINT requires 2 bytes".to_string());
327                }
328                Ok(DataValue::SmallInt(i16::from_le_bytes([
329                    bytes[0], bytes[1],
330                ])))
331            }
332            DataType::Int => {
333                if bytes.len() < 4 {
334                    return Err("INT requires 4 bytes".to_string());
335                }
336                Ok(DataValue::Int(i32::from_le_bytes([
337                    bytes[0], bytes[1], bytes[2], bytes[3],
338                ])))
339            }
340            DataType::BigInt => {
341                if bytes.len() < 8 {
342                    return Err("BIGINT requires 8 bytes".to_string());
343                }
344                Ok(DataValue::BigInt(i64::from_le_bytes([
345                    bytes[0], bytes[1], bytes[2], bytes[3], bytes[4], bytes[5], bytes[6], bytes[7],
346                ])))
347            }
348            DataType::Real => {
349                if bytes.len() < 4 {
350                    return Err("REAL requires 4 bytes".to_string());
351                }
352                Ok(DataValue::Real(OrderedF32(f32::from_le_bytes([
353                    bytes[0], bytes[1], bytes[2], bytes[3],
354                ]))))
355            }
356            DataType::DoublePrecision => {
357                if bytes.len() < 8 {
358                    return Err("DOUBLE PRECISION requires 8 bytes".to_string());
359                }
360                Ok(DataValue::DoublePrecision(OrderedF64(f64::from_le_bytes(
361                    [
362                        bytes[0], bytes[1], bytes[2], bytes[3], bytes[4], bytes[5], bytes[6],
363                        bytes[7],
364                    ],
365                ))))
366            }
367            DataType::Numeric { precision, scale } => {
368                let decoded = decode_numeric_bcd(bytes, *precision, *scale)?;
369                Ok(DataValue::Numeric(decoded))
370            }
371            DataType::Decimal { precision, scale } => {
372                let decoded = decode_numeric_bcd(bytes, *precision, *scale)?;
373                Ok(DataValue::Numeric(decoded))
374            }
375            DataType::Bool => {
376                if bytes.is_empty() {
377                    return Err("BOOLEAN requires 1 byte".to_string());
378                }
379                Ok(DataValue::Bool(bytes[0] != 0))
380            }
381            DataType::Char(n) => {
382                let len = *n as usize;
383                if bytes.len() < len {
384                    return Err(format!("CHAR({}) requires {} bytes", n, len));
385                }
386                let value = String::from_utf8(bytes[..len].to_vec())
387                    .map_err(|_| "CHAR payload is not valid UTF-8".to_string())?;
388                Ok(DataValue::Char(value))
389            }
390            DataType::Character(n) => {
391                let len = *n as usize;
392                if bytes.len() < len {
393                    return Err(format!("CHARACTER({}) requires {} bytes", n, len));
394                }
395                let value = String::from_utf8(bytes[..len].to_vec())
396                    .map_err(|_| "CHARACTER payload is not valid UTF-8".to_string())?;
397                Ok(DataValue::Char(value))
398            }
399            DataType::Varchar(max_len) => {
400                let encoded_len = ty.encoded_len(bytes)?;
401                let payload_len = u16::from_le_bytes([bytes[0], bytes[1]]) as usize;
402                if payload_len > *max_len as usize {
403                    return Err(format!(
404                        "VARCHAR payload length {} exceeds declared limit {}",
405                        payload_len, max_len
406                    ));
407                }
408                let payload = &bytes[2..encoded_len];
409                let value = String::from_utf8(payload.to_vec())
410                    .map_err(|_| "VARCHAR payload is not valid UTF-8".to_string())?;
411                Ok(DataValue::Varchar(value))
412            }
413            DataType::Date => {
414                if bytes.len() < 4 {
415                    return Err("DATE requires 4 bytes".to_string());
416                }
417                let days = i32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]) as i64;
418                let epoch = NaiveDate::from_ymd_opt(1970, 1, 1).expect("valid epoch");
419                let value = epoch
420                    .checked_add_signed(Duration::days(days))
421                    .ok_or_else(|| "DATE is outside supported chrono range".to_string())?;
422                Ok(DataValue::Date(value))
423            }
424            DataType::Time => {
425                if bytes.len() < 8 {
426                    return Err("TIME requires 8 bytes".to_string());
427                }
428                let micros = i64::from_le_bytes([
429                    bytes[0], bytes[1], bytes[2], bytes[3], bytes[4], bytes[5], bytes[6], bytes[7],
430                ]);
431                let secs = (micros / 1_000_000) as u32;
432                let nanos = ((micros % 1_000_000) * 1_000) as u32;
433                NaiveTime::from_num_seconds_from_midnight_opt(secs, nanos)
434                    .map(DataValue::Time)
435                    .ok_or_else(|| "TIME value out of range".to_string())
436            }
437            DataType::Timestamp => {
438                if bytes.len() < 8 {
439                    return Err("TIMESTAMP requires 8 bytes".to_string());
440                }
441                let micros = i64::from_le_bytes([
442                    bytes[0], bytes[1], bytes[2], bytes[3], bytes[4], bytes[5], bytes[6], bytes[7],
443                ]);
444                let epoch = NaiveDate::from_ymd_opt(1970, 1, 1)
445                    .unwrap()
446                    .and_hms_opt(0, 0, 0)
447                    .unwrap();
448                epoch
449                    .checked_add_signed(Duration::microseconds(micros))
450                    .map(DataValue::Timestamp)
451                    .ok_or_else(|| "TIMESTAMP is outside supported chrono range".to_string())
452            }
453            DataType::Bit(n) => {
454                let needed = (*n as usize).div_ceil(8);
455                if bytes.len() < needed {
456                    return Err(format!("BIT({}) requires {} bytes", n, needed));
457                }
458                Ok(DataValue::Bit(unpack_bit_string(
459                    &bytes[..needed],
460                    *n as usize,
461                )))
462            }
463        }
464    }

466    pub fn parse_and_encode(ty: &DataType, input: &str) -> Result<Vec<u8>, String> {
467        let input = input.trim();
468        validate_value(ty, input).map_err(|e| e.to_string())?;
469
470        match ty {
471            DataType::SmallInt => input
472                .parse::<i16>()
473                .map(DataValue::SmallInt)
474                .map(|v| v.to_bytes())
475                .map_err(|e| e.to_string()),
476            DataType::Int => input
477                .parse::<i32>()
478                .map(DataValue::Int)
479                .map(|v| v.to_bytes())
480                .map_err(|e| e.to_string()),
481            DataType::BigInt => input
482                .parse::<i64>()
483                .map(DataValue::BigInt)
484                .map(|v| v.to_bytes())
485                .map_err(|e| e.to_string()),
486            DataType::Real => input
487                .parse::<f32>()
488                .map(|v| DataValue::Real(OrderedF32(v)))
489                .map(|v| v.to_bytes())
490                .map_err(|e| e.to_string()),
491            DataType::DoublePrecision => input
492                .parse::<f64>()
493                .map(|v| DataValue::DoublePrecision(OrderedF64(v)))
494                .map(|v| v.to_bytes())
495                .map_err(|e| e.to_string()),
496            DataType::Numeric { precision, scale } => {
497                let parsed = parse_numeric_literal(input, *precision, *scale)?;
498                encode_numeric_bcd(&parsed, *precision)
499            }
500            DataType::Decimal { precision, scale } => {
501                let parsed = parse_numeric_literal(input, *precision, *scale)?;
502                encode_numeric_bcd(&parsed, *precision)
503            }
504            DataType::Bool => match input.to_ascii_lowercase().as_str() {
505                "true" | "t" | "1" => Ok(DataValue::Bool(true).to_bytes()),
506                "false" | "f" | "0" => Ok(DataValue::Bool(false).to_bytes()),
507                _ => Err(format!(
508                    "Invalid BOOLEAN value '{}': expected true/false",
509                    input
510                )),
511            },
512            DataType::Char(n) => {
513                let value = input.trim_matches('"').trim_matches('\'');
514                let mut bytes = value.as_bytes().to_vec();
515                if bytes.len() > *n as usize {
516                    return Err(format!("CHAR({}) value exceeds maximum length {}", n, n));
517                }
518                bytes.resize(*n as usize, b' ');
519                Ok(bytes)
520            }
521            DataType::Character(n) => {
522                let value = input.trim_matches('"').trim_matches('\'');
523                let mut bytes = value.as_bytes().to_vec();
524                if bytes.len() > *n as usize {
525                    return Err(format!(
526                        "CHARACTER({}) value exceeds maximum length {}",
527                        n, n
528                    ));
529                }
530                bytes.resize(*n as usize, b' ');
531                Ok(bytes)
532            }
533            DataType::Varchar(_) => {
534                let value = input.trim_matches('"').trim_matches('\'');
535                Ok(DataValue::Varchar(value.to_string()).to_bytes())
536            }
537            DataType::Date => {
538                let date = NaiveDate::parse_from_str(input.trim_matches('\''), "%Y-%m-%d")
539                    .map_err(|e| e.to_string())?;
540                Ok(DataValue::Date(date).to_bytes())
541            }
542            DataType::Time => {
543                let raw = input.trim_matches('\'');
544                // Accept HH:MM:SS, HH:MM:SS.ffffff
545                NaiveTime::parse_from_str(raw, "%H:%M:%S%.f")
546                    .or_else(|_| NaiveTime::parse_from_str(raw, "%H:%M:%S"))
547                    .map(|v| DataValue::Time(v).to_bytes())
548                    .map_err(|e| e.to_string())
549            }
550            DataType::Timestamp => {
551                let raw = input.trim_matches('\'');
552                NaiveDateTime::parse_from_str(raw, "%Y-%m-%d %H:%M:%S%.f")
553                    .or_else(|_| NaiveDateTime::parse_from_str(raw, "%Y-%m-%d %H:%M:%S"))
554                    .map(|v| DataValue::Timestamp(v).to_bytes())
555                    .map_err(|e| e.to_string())
556            }
557            DataType::Bit(_) => {
558                let bits = normalize_bit_literal(input);
559                Ok(DataValue::Bit(bits).to_bytes())
560            }
561        }
562    }

564    pub fn to_bytes_for_type(&self, ty: &DataType) -> Result<Vec<u8>, String> {
565        match (ty, self) {
566            (DataType::Numeric { precision, scale }, DataValue::Numeric(v)) => {
567                if v.scale != *scale {
568                    return Err(format!(
569                        "NUMERIC scale mismatch: value has scale {}, type requires {}",
570                        v.scale, scale
571                    ));
572                }
573                encode_numeric_bcd(v, *precision)
574            }
575            (DataType::Decimal { precision, scale }, DataValue::Numeric(v)) => {
576                if v.scale != *scale {
577                    return Err(format!(
578                        "DECIMAL scale mismatch: value has scale {}, type requires {}",
579                        v.scale, scale
580                    ));
581                }
582                encode_numeric_bcd(v, *precision)
583            }
584            (DataType::Char(n), DataValue::Char(v)) => {
585                let mut bytes = v.as_bytes().to_vec();
586                if bytes.len() > *n as usize {
587                    return Err(format!("CHAR({}) value exceeds maximum length {}", n, n));
588                }
589                bytes.resize(*n as usize, b' ');
590                Ok(bytes)
591            }
592            (DataType::Character(n), DataValue::Char(v)) => {
593                let mut bytes = v.as_bytes().to_vec();
594                if bytes.len() > *n as usize {
595                    return Err(format!(
596                        "CHARACTER({}) value exceeds maximum length {}",
597                        n, n
598                    ));
599                }
600                bytes.resize(*n as usize, b' ');
601                Ok(bytes)
602            }
603            _ => Ok(self.to_bytes()),
