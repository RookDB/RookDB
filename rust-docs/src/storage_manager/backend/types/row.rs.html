9fn data_value_matches_type(ty: &DataType, value: &DataValue) -> bool {
10    matches!(
11        (ty, value),
12        (DataType::SmallInt, DataValue::SmallInt(_))
13            | (DataType::Int, DataValue::Int(_))
14            | (DataType::BigInt, DataValue::BigInt(_))
15            | (DataType::Real, DataValue::Real(_))
16            | (DataType::DoublePrecision, DataValue::DoublePrecision(_))
17            | (DataType::Numeric { .. }, DataValue::Numeric(_))
18            | (DataType::Decimal { .. }, DataValue::Numeric(_))
19            | (DataType::Bool, DataValue::Bool(_))
20            | (DataType::Char(_), DataValue::Char(_))
21            | (DataType::Character(_), DataValue::Char(_))
22            | (DataType::Varchar(_), DataValue::Varchar(_))
23            | (DataType::Date, DataValue::Date(_))
24            | (DataType::Time, DataValue::Time(_))
25            | (DataType::Bit(_), DataValue::Bit(_))
26            | (DataType::Timestamp, DataValue::Timestamp(_))
27    )
28}

32/// Returns the exact on-disk byte size of a row given a schema and concrete
33/// values, without constructing a `Row` or serialising anything.
34/// Useful for pre-insert capacity checks in the heap/page manager.
35pub fn row_byte_size(schema: &[DataType], values: &[Option<&DataValue>]) -> usize {
36    let physical = PhysicalSchema::from_logical(schema);
37    let layout = RowLayout::compute(&physical);
38    let mut size = layout.min_row_size();
39    for (k, &log_idx) in physical.varlen_indices_logical.iter().enumerate() {
40        if let Some(Some(val)) = values.get(log_idx) {
41            let bytes = val.to_bytes_for_type(&schema[log_idx]).unwrap_or_default();
42            // stored bytes for varchar are exclusively raw payload
43            let payload_len = bytes.len();
44            let _ = k; // offset-table slot already counted in min_row_size
45            size += payload_len;
46        }
47    }
48    size
49}

53/// Encode a row from string literals.
54/// Layout: [Header 4B][Null Bitmap][Var-Len Offset Table][Fixed Data][Var-Len Data]
55/// Var-len payloads are stored **without** a length prefix; lengths are derived
56/// from adjacent offset-table entries and the total row size.
57pub fn serialize_nullable_row(
58    schema: &[DataType],
59    values: &[Option<&str>],
60) -> Result<Vec<u8>, String> {
61    if schema.len() != values.len() {
62        return Err(format!(
63            "Schema/value length mismatch: schema={}, values={}",
64            schema.len(),
65            values.len()
66        ));
67    }
68
69    // Pre-encode all values so we know payload sizes upfront.
70    let mut encoded_values: Vec<Option<Vec<u8>>> = Vec::with_capacity(schema.len());
71    let mut bitmap = NullBitmap::new(schema.len());
72
73    for (i, (ty, maybe_raw)) in schema.iter().zip(values.iter()).enumerate() {
74        match maybe_raw {
75            Some(raw) => {
76                let enc = DataValue::parse_and_encode(ty, raw)?;
77                encoded_values.push(Some(enc));
78            }
79            None => {
80                bitmap.set_null(i);
81                encoded_values.push(None);
82            }
83        }
84    }
85
86    serialize_encoded(schema, &encoded_values, &bitmap)
87}

89/// Encode a row from typed `DataValue` instances.
90pub fn serialize_nullable_typed_row(
91    schema: &[DataType],
92    values: &[Option<DataValue>],
93) -> Result<Vec<u8>, String> {
94    if schema.len() != values.len() {
95        return Err(format!(
96            "Schema/value length mismatch: schema={}, values={}",
97            schema.len(),
98            values.len()
99        ));
100    }
101
102    let mut encoded_values: Vec<Option<Vec<u8>>> = Vec::with_capacity(schema.len());
103    let mut bitmap = NullBitmap::new(schema.len());
104
105    for (i, (ty, maybe_val)) in schema.iter().zip(values.iter()).enumerate() {
106        match maybe_val {
107            Some(val) => {
108                if !data_value_matches_type(ty, val) {
109                    return Err(format!(
110                        "Type mismatch at column {}: expected {}, got {}",
111                        i,
112                        ty,
113                        value_type_name(val)
114                    ));
115                }
116                encoded_values.push(Some(val.to_bytes_for_type(ty)?));
117            }
118            None => {
119                bitmap.set_null(i);
120                encoded_values.push(None);
121            }
122        }
123    }
124
125    serialize_encoded(schema, &encoded_values, &bitmap)
126}

128/// Core serialization path — shared by both public entry points.
129///
130/// `encoded_values[i]` is the raw bytes produced by the type's encoder:
131/// - Fixed-length types: exactly `fixed_size()` bytes.
132/// - Varchar: Raw UTF-8 payload bytes exclusively.
133fn serialize_encoded(
134    schema: &[DataType],
135    encoded_values: &[Option<Vec<u8>>],
136    bitmap: &NullBitmap,
137) -> Result<Vec<u8>, String> {
138    let physical = PhysicalSchema::from_logical(schema);
139    let layout = RowLayout::compute(&physical);
140
141    // Compute total var-len payload size (raw bytes, no prefix).
142    let varlen_payload_total: usize = physical
143        .varlen_indices_logical
144        .iter()
145        .map(|&log_idx| {
146            encoded_values[log_idx]
147                .as_ref()
148                .map(|enc| enc.len())
149                .unwrap_or(0)
150        })
151        .sum();
152
153    let total_size = layout.min_row_size() + varlen_payload_total;
154    let mut buf = vec![0u8; total_size];
155
156    // ── Header ────────────────────────────────────────────────────────────────
157    let num_cols = schema.len() as u16;
158    let num_varlen = physical.num_varlen() as u16;
159    buf[0..2].copy_from_slice(&num_cols.to_le_bytes());
160    buf[2..4].copy_from_slice(&num_varlen.to_le_bytes());
161
162    // ── Null bitmap ───────────────────────────────────────────────────────────
163    let bm_start = RowLayout::bitmap_offset();
164    buf[bm_start..bm_start + layout.null_bitmap_size].copy_from_slice(bitmap.as_bytes());
165
166    // ── Var-len offset table (initialised to 0x0000) ──────────────────────────
167    // (Already zeroed by vec![0u8; …])
168
169    // ── Fixed-length data region ───────────────────────────────────────────────
170    for (rank, &log_idx) in physical.fixed_indices_logical.iter().enumerate() {
171        if let Some(enc) = &encoded_values[log_idx] {
172            let col_start = layout.fixed_data_start + layout.fixed_col_offsets[rank];
173            buf[col_start..col_start + enc.len()].copy_from_slice(enc);
174        }
175        // NULL → bytes stay 0x00 (correct; bitmap is the authority)
176    }
177
178    // ── Var-len data region + fill offset table ────────────────────────────────
179    let vt_start = layout.varlen_table_offset();
180    let mut append_cursor = layout.min_row_size(); // write pointer into buf tail
181
182    for (vl_rank, &log_idx) in physical.varlen_indices_logical.iter().enumerate() {
183        let offset_slot = vt_start + vl_rank * 2;
184        if let Some(payload) = &encoded_values[log_idx] {
185            // Record offset of this payload from row start
186            let row_offset = append_cursor as u16;
187            buf[offset_slot..offset_slot + 2].copy_from_slice(&row_offset.to_le_bytes());
188
189            buf[append_cursor..append_cursor + payload.len()].copy_from_slice(payload);
190            append_cursor += payload.len();
191        }
192        // NULL → offset table stays 0x0000 (bitmap is the authority)
193    }
194
195    Ok(buf)
196}

198/// Decode a serialized row back into typed values (logical order).
199/// `total_row_size` must be the byte length of `row_bytes` (available from
200/// the slot directory); it is used to derive the last var-len payload length.
201pub fn deserialize_nullable_row(
202    schema: &[DataType],
203    row_bytes: &[u8],
204) -> Result<Vec<Option<DataValue>>, String> {
205    let total_row_size = row_bytes.len();
206
207    if total_row_size < 4 {
208        return Err("Row too short to contain header".to_string());
209    }
210
211    // ── Header ────────────────────────────────────────────────────────────────
212    let num_cols_stored = u16::from_le_bytes([row_bytes[0], row_bytes[1]]) as usize;
213    let num_varlen_stored = u16::from_le_bytes([row_bytes[2], row_bytes[3]]) as usize;
214
215    if num_cols_stored != schema.len() {
216        return Err(format!(
217            "Header column count {} does not match schema length {}",
218            num_cols_stored,
219            schema.len()
220        ));
221    }
222
223    let physical = PhysicalSchema::from_logical(schema);
224    let layout = RowLayout::compute(&physical);
225
226    if num_varlen_stored != physical.num_varlen() {
227        return Err(format!(
228            "Header var-len count {} does not match schema var-len count {}",
229            num_varlen_stored,
230            physical.num_varlen()
231        ));
232    }
233
234    if total_row_size < layout.min_row_size() {
235        return Err(format!(
236            "Row ({} bytes) shorter than minimum layout size ({} bytes)",
237            total_row_size,
238            layout.min_row_size()
239        ));
240    }
241
242    // ── Null bitmap ───────────────────────────────────────────────────────────
243    let bm_start = RowLayout::bitmap_offset();
244    let bitmap = NullBitmap::from_bytes(
245        schema.len(),
246        &row_bytes[bm_start..bm_start + layout.null_bitmap_size],
247    )?;
248
249    // ── Var-len offset table ───────────────────────────────────────────────────
250    let vt_start = layout.varlen_table_offset();
251    let mut varlen_offsets: Vec<Option<usize>> = Vec::with_capacity(physical.num_varlen());
252
253    for vl_rank in 0..physical.num_varlen() {
254        let slot = vt_start + vl_rank * 2;
255        let raw = u16::from_le_bytes([row_bytes[slot], row_bytes[slot + 1]]) as usize;
256        // 0x0000 = NULL sentinel; bitmap is authoritative but we read the table anyway
257        varlen_offsets.push(if raw == 0 { None } else { Some(raw) });
258    }
259
260    // Build a lookup: varlen physical-group rank → row-byte range
261    // length[k] = offset[k+1] - offset[k];  length[last] = total_row_size - offset[last]
262    let mut varlen_ranges: Vec<Option<(usize, usize)>> = vec![None; physical.num_varlen()];
263
264    // Collect non-null offsets with their ranks for range computation
265    let non_null_slots: Vec<(usize, usize)> = varlen_offsets
266        .iter()
267        .enumerate()
268        .filter_map(|(rank, opt)| opt.map(|off| (rank, off)))
269        .collect();
270
271    for (i, &(rank, start)) in non_null_slots.iter().enumerate() {
272        let end = if i + 1 < non_null_slots.len() {
273            non_null_slots[i + 1].1 // start of next non-null var-len col
274        } else {
275            total_row_size // last payload extends to end of row
276        };
277        varlen_ranges[rank] = Some((start, end));
278    }
279
280    // ── Decode columns in logical order ───────────────────────────────────────
281    let mut out: Vec<Option<DataValue>> = vec![None; schema.len()];
282
283    for log_idx in 0..schema.len() {
284        if bitmap.is_null(log_idx) {
285            out[log_idx] = None;
286            continue;
287        }
288
289        let ty = &schema[log_idx];
290        let phys_idx = physical.logical_to_physical[log_idx];
291
292        if ty.is_fixed_length() {
293            // rank within the fixed group = phys_idx (fixed cols occupy 0..n_fixed)
294            let rank = phys_idx;
295            let col_start = layout.fixed_data_start + layout.fixed_col_offsets[rank];
296            let col_size =
297                ty.fixed_size()
298                    .expect("fixed-length type must have fixed_size") as usize;
299            let value = DataValue::from_bytes(ty, &row_bytes[col_start..col_start + col_size])?;
300            out[log_idx] = Some(value);
301        } else {
302            // var-len column — rank within the var-len group
303            let vl_rank = phys_idx - physical.num_fixed();
304            if let Some((start, end)) = varlen_ranges[vl_rank] {
305                // Pass exact payload bytes to the decoder
306                let payload = &row_bytes[start..end];
307                let value = DataValue::from_bytes(ty, payload)?;
308                out[log_idx] = Some(value);
309            }
310            // else: offset was 0x0000 but bitmap said non-null — treat as null
311        }
312    }
313
314    Ok(out)
315}

319#[derive(Debug, Clone)]
320pub struct Row {
321    /// User-defined column order (logical schema). Public API indices use this.
322    logical_schema: Vec<DataType>,
323    /// On-disk column reordering (fixed-first, then var-len).
324    physical: PhysicalSchema,
325    /// Byte-level geometry derived from `physical`.
326    layout: RowLayout,
327    /// NULL flags keyed on **logical** column index.
328    null_bitmap: NullBitmap,
329    /// On-disk bytes: [var-len offset table][fixed-data][var-len data].
330    /// Does NOT include the 4-byte header or null bitmap (those are prepended
331    /// by `serialize()`).
332    data: Vec<u8>,
333}
334
335impl Row {
336    /// Create a new all-NULL row for the given schema.
337    pub fn new(schema: Vec<DataType>) -> Self {
338        let physical = PhysicalSchema::from_logical(&schema);
339        let layout = RowLayout::compute(&physical);
340        let mut null_bitmap = NullBitmap::new(schema.len());
341        for i in 0..schema.len() {
342            null_bitmap.set_null(i);
343        }
344        // Pre-allocate the static regions (var-len table + fixed data) all zeroed.
345        let data = vec![0u8; layout.varlen_table_size + layout.fixed_data_size];
346        Self {
347            logical_schema: schema,
348            physical,
349            layout,
350            null_bitmap,
351            data,
352        }
353    }

357    /// Returns the exact on-disk byte size of this row without serialising it.
358    /// Used by the heap/page manager to check if a row fits before writing.
359    pub fn byte_size(&self) -> usize {
360        RowLayout::bitmap_offset() + self.layout.null_bitmap_size + self.data.len()
361    }

365    pub fn set_value(&mut self, column_index: usize, value: &DataValue) -> Result<(), String> {
366        if column_index >= self.logical_schema.len() {
367            return Err(format!("Column index {} out of bounds", column_index));
368        }
369        let ty = &self.logical_schema[column_index];
370        if !data_value_matches_type(ty, value) {
371            return Err(format!(
372                "Type mismatch at column {}: expected {}, got {}",
373                column_index,
374                ty,
375                value_type_name(value)
376            ));
377        }
378
379        let enc = value.to_bytes_for_type(ty)?;
380        let phys_idx = self.physical.logical_to_physical[column_index];
381        let bm_start = RowLayout::bitmap_offset();
382        let data_start_in_row = bm_start + self.layout.null_bitmap_size;
383
384        self.null_bitmap.clear_null(column_index);
385
386        if ty.is_fixed_length() {
387            let rank = phys_idx;
388            let row_col_start = self.layout.fixed_data_start + self.layout.fixed_col_offsets[rank];
389            let self_col_start = row_col_start - data_start_in_row;
390            self.data[self_col_start..self_col_start + enc.len()].copy_from_slice(&enc);
391            return Ok(());
392        }
393
394        let vl_rank = phys_idx - self.physical.num_fixed();
395        let (payload_start, payload_end) = self.get_varlen_bounds(vl_rank);
396        let old_len = payload_end - payload_start;
397        let new_len = enc.len();
398
399        // Ensure its offset is active
400        let slot = vl_rank * 2;
401        let mut raw_offset = u16::from_le_bytes([self.data[slot], self.data[slot + 1]]) as usize;
402
403        if raw_offset == 0 {
404            raw_offset = payload_start + data_start_in_row;
405            self.data[slot..slot + 2].copy_from_slice(&(raw_offset as u16).to_le_bytes());
406        }
407
408        if new_len > old_len {
409            self.data.splice(payload_start..payload_end, enc);
410            self.adjust_varlen_offsets(vl_rank, (new_len - old_len) as isize);
411        } else if new_len < old_len {
412            self.data.splice(payload_start..payload_end, enc);
413            self.adjust_varlen_offsets(vl_rank, -((old_len - new_len) as isize));
414        } else {
415            self.data[payload_start..payload_end].copy_from_slice(&enc);
416        }
417
418        Ok(())
419    }

421    pub fn set_null(&mut self, column_index: usize) -> Result<(), String> {
422        if column_index >= self.logical_schema.len() {
423            return Err(format!("Column index {} out of bounds", column_index));
424        }
425
426        if self.null_bitmap.is_null(column_index) {
427            return Ok(()); // Already NULL
428        }
429
430        let ty = &self.logical_schema[column_index];
431        let phys_idx = self.physical.logical_to_physical[column_index];
432        self.null_bitmap.set_null(column_index);
433
434        if ty.is_fixed_length() {
435            return Ok(()); // Zero dead space handling needed for fixed
436        }
437
438        let vl_rank = phys_idx - self.physical.num_fixed();
439        let (payload_start, payload_end) = self.get_varlen_bounds(vl_rank);
440        let bytes_to_remove = payload_end - payload_start;
441
442        if bytes_to_remove > 0 {
443            self.data.drain(payload_start..payload_end);
444            self.adjust_varlen_offsets(vl_rank, -(bytes_to_remove as isize));
445        }
446
447        // Set offset to 0 sentinel
448        let slot = vl_rank * 2;
449        self.data[slot..slot + 2].copy_from_slice(&0u16.to_le_bytes());
450
451        Ok(())
452    }

454    pub fn get_value(&self, column_index: usize) -> Result<Option<DataValue>, String> {
455        if column_index >= self.logical_schema.len() {
456            return Err(format!("Column index {} out of bounds", column_index));
457        }
458
459        if self.null_bitmap.is_null(column_index) {
460            return Ok(None);
461        }
462
463        let ty = &self.logical_schema[column_index];
464        let phys_idx = self.physical.logical_to_physical[column_index];
465
466        let bm_start = RowLayout::bitmap_offset();
467        let data_start_in_row = bm_start + self.layout.null_bitmap_size;
468
469        if ty.is_fixed_length() {
470            let rank = phys_idx;
471            let row_col_start = self.layout.fixed_data_start + self.layout.fixed_col_offsets[rank];
472            let self_col_start = row_col_start - data_start_in_row;
473
474            let col_size =
475                ty.fixed_size()
476                    .expect("fixed-length type must have fixed_size") as usize;
477            let bytes = &self.data[self_col_start..self_col_start + col_size];
478            return Ok(Some(DataValue::from_bytes(ty, bytes)?));
479        }
480
481        let vl_rank = phys_idx - self.physical.num_fixed();
482        let (start, end) = self.get_varlen_bounds(vl_rank);
483        let bytes = &self.data[start..end];
484        Ok(Some(DataValue::from_bytes(ty, bytes)?))
485    }

489    /// Produce the full on-disk byte sequence.
490    pub fn serialize(&self) -> Vec<u8> {
491        let bm_start = RowLayout::bitmap_offset();
492        let data_start = bm_start + self.layout.null_bitmap_size;
493        let total = data_start + self.data.len();
494
495        let mut out = vec![0u8; total];
496        // Header
497        out[0..2].copy_from_slice(&(self.logical_schema.len() as u16).to_le_bytes());
498        out[2..4].copy_from_slice(&(self.physical.num_varlen() as u16).to_le_bytes());
499        // Null bitmap
500        out[bm_start..bm_start + self.layout.null_bitmap_size]
501            .copy_from_slice(self.null_bitmap.as_bytes());
502        // Remaining data (var-len table + fixed + var-len payloads)
503        out[data_start..].copy_from_slice(&self.data);
504        out
505    }

507    pub fn deserialize(schema: &[DataType], bytes: &[u8]) -> Result<Self, String> {
508        let physical = PhysicalSchema::from_logical(schema);
509        let layout = RowLayout::compute(&physical);
510
511        if bytes.len() < 4 {
512            return Err("Row too short to contain header".to_string());
513        }
514
515        // Validate via full decode first (catches corrupt rows early).
516        let _ = deserialize_nullable_row(schema, bytes)?;
517
518        // Extract null bitmap and raw data from the byte slice.
519        let bm_start = RowLayout::bitmap_offset();
520        let null_bitmap = NullBitmap::from_bytes(
521            schema.len(),
522            &bytes[bm_start..bm_start + layout.null_bitmap_size],
523        )?;
524
525        // `data` = everything after the header and null bitmap
526        let data_start = bm_start + layout.null_bitmap_size;
527        let data = bytes[data_start..].to_vec();
528
529        Ok(Self {
530            logical_schema: schema.to_vec(),
531            physical,
532            layout,
533            null_bitmap,
534            data,
535        })
536    }

538    // ── Internal helpers ──────────────────────────────────────────────────────
539
540    /// Returns the absolute `(start, end)` offset bounds for a var-len payload
541    /// inside `self.data`. Uses exactly the same logic as deserialize, ensuring
542    /// we can splice values optimally without breaking layout geometry.
543    fn get_varlen_bounds(&self, vl_rank: usize) -> (usize, usize) {
544        let bm_start = RowLayout::bitmap_offset();
545        let data_start_in_row = bm_start + self.layout.null_bitmap_size;
546
547        let slot = vl_rank * 2;
548        let raw_offset = u16::from_le_bytes([self.data[slot], self.data[slot + 1]]) as usize;
549
550        if raw_offset == 0 {
551            // Null column: its theoretical insertion point is the beginning
552            // of the next sequence.
553            let mut insertion_point = self.data.len();
554            for r in (vl_rank + 1)..self.physical.num_varlen() {
555                let next_slot = r * 2;
556                let next_off =
557                    u16::from_le_bytes([self.data[next_slot], self.data[next_slot + 1]]) as usize;
558                if next_off != 0 {
559                    insertion_point = next_off - data_start_in_row;
560                    break;
561                }
562            }
563            return (insertion_point, insertion_point);
564        }
565
566        let start = raw_offset - data_start_in_row;
567        let mut end = self.data.len();
568        for r in (vl_rank + 1)..self.physical.num_varlen() {
569            let next_slot = r * 2;
570            let next_off =
571                u16::from_le_bytes([self.data[next_slot], self.data[next_slot + 1]]) as usize;
572            if next_off != 0 {
573                end = next_off - data_start_in_row;
574                break;
575            }
576        }
577
578        (start, end)
579    }

581    /// Shift the absolute offset reference for all variable-length rows mapped
582    /// geometrically *after* the given rank up or down by `delta`.
583    fn adjust_varlen_offsets(&mut self, after_vl_rank: usize, delta: isize) {
584        for r in (after_vl_rank + 1)..self.physical.num_varlen() {
585            let slot = r * 2;
586            let current_off = u16::from_le_bytes([self.data[slot], self.data[slot + 1]]) as usize;
587            if current_off != 0 {
588                let updated = (current_off as isize + delta) as u16;
589                self.data[slot..slot + 2].copy_from_slice(&updated.to_le_bytes());
590            }
591        }
