6pub fn serialize_nullable_row(
7    schema: &[DataType],
8    values: &[Option<&str>],
9) -> Result<Vec<u8>, String> {
10    if schema.len() != values.len() {
11        return Err(format!(
12            "Schema/value length mismatch: schema={}, values={}",
13            schema.len(),
14            values.len()
15        ));
16    }
17
18    let mut bitmap = NullBitmap::new(schema.len());
19    let bitmap_len = schema.len().div_ceil(8);
20    let mut row = vec![0u8; bitmap_len];
21
22    for (i, (ty, maybe_value)) in schema.iter().zip(values.iter()).enumerate() {
23        match maybe_value {
24            Some(raw) => {
25                let encoded = DataValue::parse_and_encode(ty, raw)?;
26                row.extend_from_slice(&encoded);
27            }
28            None => bitmap.set_null(i),
29        }
30    }
31
32    row[..bitmap_len].copy_from_slice(bitmap.as_bytes());
33    Ok(row)
34}

36pub fn deserialize_nullable_row(
37    schema: &[DataType],
38    row_bytes: &[u8],
39) -> Result<Vec<Option<DataValue>>, String> {
40    let bitmap_len = schema.len().div_ceil(8);
41    if row_bytes.len() < bitmap_len {
42        return Err("Row shorter than NULL bitmap".to_string());
43    }
44
45    let bitmap = NullBitmap::from_bytes(schema.len(), &row_bytes[..bitmap_len])?;
46    let mut cursor = bitmap_len;
47    let mut out = Vec::with_capacity(schema.len());
48
49    for (i, ty) in schema.iter().enumerate() {
50        if bitmap.is_null(i) {
51            out.push(None);
52            continue;
53        }
54
55        let remaining = &row_bytes[cursor..];
56        let encoded_len = ty.encoded_len(remaining)?;
57        let value = DataValue::from_bytes(ty, &remaining[..encoded_len])?;
58        cursor += encoded_len;
59        out.push(Some(value));
60    }
61
62    if cursor != row_bytes.len() {
63        return Err(format!(
64            "Row has {} trailing byte(s) after decode",
65            row_bytes.len() - cursor
66        ));
67    }
68
69    Ok(out)
70}

72fn data_value_matches_type(ty: &DataType, value: &DataValue) -> bool {
73    matches!(
74        (ty, value),
75        (DataType::SmallInt, DataValue::SmallInt(_))
76            | (DataType::Int, DataValue::Int(_))
77            | (DataType::BigInt, DataValue::BigInt(_))
78            | (DataType::Real, DataValue::Real(_))
79            | (DataType::DoublePrecision, DataValue::DoublePrecision(_))
80            | (DataType::Numeric { .. }, DataValue::Numeric(_))
81            | (DataType::Decimal { .. }, DataValue::Numeric(_))
82            | (DataType::Bool, DataValue::Bool(_))
83            | (DataType::Char(_), DataValue::Char(_))
84            | (DataType::Character(_), DataValue::Char(_))
85            | (DataType::Varchar(_), DataValue::Varchar(_))
86            | (DataType::Date, DataValue::Date(_))
87            | (DataType::Time, DataValue::Time(_))
88            | (DataType::Bit(_), DataValue::Bit(_))
89            | (DataType::Timestamp, DataValue::Timestamp(_))
90    )
91}

93fn serialize_nullable_typed_row(
94    schema: &[DataType],
95    values: &[Option<DataValue>],
96) -> Result<Vec<u8>, String> {
97    if schema.len() != values.len() {
98        return Err(format!(
99            "Schema/value length mismatch: schema={}, values={}",
100            schema.len(),
101            values.len()
102        ));
103    }
104
105    let bitmap_len = schema.len().div_ceil(8);
106    let mut bitmap = NullBitmap::new(schema.len());
107    let mut out = vec![0u8; bitmap_len];
108
109    for (i, (ty, maybe_value)) in schema.iter().zip(values.iter()).enumerate() {
110        match maybe_value {
111            Some(value) => {
112                if !data_value_matches_type(ty, value) {
113                    return Err(format!(
114                        "Type mismatch at column {}: expected {}, got {}",
115                        i,
116                        ty,
117                        value_type_name(value)
118                    ));
119                }
120                let encoded = value.to_bytes_for_type(ty)?;
121                out.extend_from_slice(&encoded);
122            }
123            None => bitmap.set_null(i),
124        }
125    }
126
127    out[..bitmap_len].copy_from_slice(bitmap.as_bytes());
128    Ok(out)
129}

131#[derive(Debug, Clone)]
132pub struct Row {
133    schema: Vec<DataType>,
134    null_bitmap: NullBitmap,
135    data: Vec<u8>,
136}
137
138impl Row {
139    pub fn new(schema: Vec<DataType>) -> Self {
140        let mut null_bitmap = NullBitmap::new(schema.len());
141        for i in 0..schema.len() {
142            null_bitmap.set_null(i);
143        }
144        Self {
145            schema,
146            null_bitmap,
147            data: Vec::new(),
148        }
149    }
150
151    fn to_values(&self) -> Result<Vec<Option<DataValue>>, String> {
152        deserialize_nullable_row(&self.schema, &self.serialize())
153    }
154
155    fn rebuild_from_values(&mut self, values: &[Option<DataValue>]) -> Result<(), String> {
156        let row_bytes = serialize_nullable_typed_row(&self.schema, values)?;
157        let bitmap_len = self.schema.len().div_ceil(8);
158        self.null_bitmap = NullBitmap::from_bytes(self.schema.len(), &row_bytes[..bitmap_len])?;
159        self.data = row_bytes[bitmap_len..].to_vec();
160        Ok(())
161    }
162
163    pub fn set_value(&mut self, column_index: usize, value: &DataValue) -> Result<(), String> {
164        if column_index >= self.schema.len() {
165            return Err(format!("Column index {} out of bounds", column_index));
166        }
167
168        if !data_value_matches_type(&self.schema[column_index], value) {
169            return Err(format!(
170                "Type mismatch at column {}: expected {}, got {}",
171                column_index,
172                self.schema[column_index],
173                value_type_name(value)
174            ));
175        }
176
177        let mut values = self.to_values()?;
178        values[column_index] = Some(value.clone());
179        self.rebuild_from_values(&values)
180    }
181
182    pub fn set_null(&mut self, column_index: usize) -> Result<(), String> {
183        if column_index >= self.schema.len() {
184            return Err(format!("Column index {} out of bounds", column_index));
185        }
186
187        let mut values = self.to_values()?;
188        values[column_index] = None;
189        self.rebuild_from_values(&values)
190    }
191
192    pub fn get_value(&self, column_index: usize) -> Result<Option<DataValue>, String> {
193        if column_index >= self.schema.len() {
194            return Err(format!("Column index {} out of bounds", column_index));
195        }
196        let values = self.to_values()?;
197        Ok(values[column_index].clone())
198    }
199
200    pub fn serialize(&self) -> Vec<u8> {
201        let bitmap_len = self.schema.len().div_ceil(8);
202        let mut out = Vec::with_capacity(bitmap_len + self.data.len());
203        out.extend_from_slice(self.null_bitmap.as_bytes());
204        out.extend_from_slice(&self.data);
205        out
206    }
207
208    pub fn deserialize(schema: &[DataType], bytes: &[u8]) -> Result<Self, String> {
209        let bitmap_len = schema.len().div_ceil(8);
210        if bytes.len() < bitmap_len {
211            return Err("Row shorter than NULL bitmap".to_string());
212        }
213
214        let _ = deserialize_nullable_row(schema, bytes)?;
215
216        let null_bitmap = NullBitmap::from_bytes(schema.len(), &bytes[..bitmap_len])?;
217        let data = bytes[bitmap_len..].to_vec();
218        Ok(Self {
219            schema: schema.to_vec(),
220            null_bitmap,
221            data,
222        })
223    }
224}
