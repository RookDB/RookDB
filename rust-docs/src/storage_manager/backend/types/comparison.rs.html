1//! SQL-99 comparison semantics for typed values.
2//!
3//! Provides:
4//! - [`ComparisonError`] — the typed error returned for incompatible types.
5//! - [`Comparable`] — the comparison trait implemented by [`DataValue`].
6//! - [`compare_nullable`] / [`nullable_equals`] — NULL-aware wrappers that
7//!   return `None` (UNKNOWN) when either operand is NULL, matching SQL semantics.
8
9use std::cmp::Ordering;
10use std::fmt;
11
12use crate::types::value::DataValue;

14// ── Error type ────────────────────────────────────────────────────────────────
15
16/// Error returned when two incompatible types are compared.
17#[derive(Debug, Clone, PartialEq, Eq)]
18pub enum ComparisonError {
19    /// The left and right operands have types that cannot be compared.
20    TypeMismatch { left: String, right: String },
21}
22
23impl fmt::Display for ComparisonError {
24    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
25        match self {
26            ComparisonError::TypeMismatch { left, right } => {
27                write!(f, "Cannot compare {} with {}", left, right)
28            }
29        }
30    }
31}
32
33impl std::error::Error for ComparisonError {}
34
35// ── Comparable trait ──────────────────────────────────────────────────────────
36
37/// Trait for types that support SQL-style ordered comparison.
38///
39/// Numeric types (`SMALLINT`, `INT`, `BIGINT`) are mutually comparable via
40/// implicit widening promotion. All other types are only comparable to
41/// themselves. Cross-type comparisons return [`ComparisonError::TypeMismatch`].
42pub trait Comparable: Sized {
43    /// Compare `self` to `other`, returning the ordering relationship.
44    fn compare(&self, other: &Self) -> Result<Ordering, ComparisonError>;
45
46    /// Returns `true` if `self` and `other` are equal under SQL semantics.
47    fn is_equal(&self, other: &Self) -> Result<bool, ComparisonError> {
48        Ok(self.compare(other)? == Ordering::Equal)
49    }
50}

54/// Returns a static SQL type name string for a `DataValue` variant.
55/// Used in error messages to identify the type of an unexpected value.
56pub(crate) fn value_type_name(value: &DataValue) -> &'static str {
57    match value {
58        DataValue::SmallInt(_) => "SMALLINT",
59        DataValue::Int(_) => "INT",
60        DataValue::BigInt(_) => "BIGINT",
61        DataValue::Real(_) => "REAL",
62        DataValue::DoublePrecision(_) => "DOUBLE PRECISION",
63        DataValue::Numeric(_) => "NUMERIC",
64        DataValue::Bool(_) => "BOOLEAN",
65        DataValue::Char(_) => "CHAR",
66        DataValue::Varchar(_) => "VARCHAR",
67        DataValue::Date(_) => "DATE",
68        DataValue::Time(_) => "TIME",
69        DataValue::Bit(_) => "BIT",
70        DataValue::Timestamp(_) => "TIMESTAMP",
71    }
72}

74// ── DataValue comparison ──────────────────────────────────────────────────────
75
76impl Comparable for DataValue {
77    /// Compare two `DataValue` instances.
78    ///
79    /// **Numeric promotion:** `SMALLINT` ↔ `INT` ↔ `BIGINT` comparisons are
80    /// supported by widening the narrower operand before comparing.
81    ///
82    /// **NUMERIC/DECIMAL:** compared by normalising scales before comparing
83    /// the unscaled integer representations.
84    ///
85    /// **CHAR:** trailing spaces are stripped before lexicographic comparison,
86    /// matching SQL padding-ignoring semantics.
87    ///
88    /// All other cross-type combinations return [`ComparisonError::TypeMismatch`].
89    fn compare(&self, other: &Self) -> Result<Ordering, ComparisonError> {
90        match (self, other) {
91            // Same-type integer comparisons
92            (DataValue::SmallInt(a), DataValue::SmallInt(b)) => Ok(a.cmp(b)),
93            (DataValue::Int(a), DataValue::Int(b)) => Ok(a.cmp(b)),
94            (DataValue::BigInt(a), DataValue::BigInt(b)) => Ok(a.cmp(b)),
95
96            // Cross-width integer promotion (widen to the larger type)
97            (DataValue::SmallInt(a), DataValue::Int(b)) => Ok((*a as i32).cmp(b)),
98            (DataValue::Int(a), DataValue::SmallInt(b)) => Ok(a.cmp(&(*b as i32))),
99            (DataValue::SmallInt(a), DataValue::BigInt(b)) => Ok((*a as i64).cmp(b)),
100            (DataValue::BigInt(a), DataValue::SmallInt(b)) => Ok(a.cmp(&(*b as i64))),
101            (DataValue::Int(a), DataValue::BigInt(b)) => Ok((*a as i64).cmp(b)),
102            (DataValue::BigInt(a), DataValue::Int(b)) => Ok(a.cmp(&(*b as i64))),
103
104            // Floating-point (NaN-aware via OrderedF32/OrderedF64)
105            (DataValue::Real(a), DataValue::Real(b)) => Ok(a.cmp(b)),
106            (DataValue::DoublePrecision(a), DataValue::DoublePrecision(b)) => Ok(a.cmp(b)),
107
108            // Exact decimal — normalise scales before comparing unscaled values
109            (DataValue::Numeric(a), DataValue::Numeric(b)) => {
110                let ordering = if a.scale == b.scale {
111                    // Same scale: compare unscaled integers directly
112                    a.unscaled.cmp(&b.unscaled)
113                } else if a.scale > b.scale {
114                    // a has more fractional digits; scale up b to match
115                    let factor = 10_i128.pow((a.scale - b.scale) as u32);
116                    a.unscaled.cmp(&(b.unscaled * factor))
117                } else {
118                    // b has more fractional digits; scale up a to match
119                    let factor = 10_i128.pow((b.scale - a.scale) as u32);
120                    (a.unscaled * factor).cmp(&b.unscaled)
121                };
122                Ok(ordering)
123            }
124
125            // Boolean: false < true
126            (DataValue::Bool(a), DataValue::Bool(b)) => Ok(a.cmp(b)),
127
128            // String types
129            // CHAR: ignore trailing spaces (SQL padding-ignoring semantics)
130            (DataValue::Char(a), DataValue::Char(b)) => Ok(a.trim_end().cmp(b.trim_end())),
131            // VARCHAR: exact byte-wise lexicographic comparison
132            (DataValue::Varchar(a), DataValue::Varchar(b)) => Ok(a.cmp(b)),
133
134            // Temporal types: chronological ordering
135            (DataValue::Date(a), DataValue::Date(b)) => Ok(a.cmp(b)),
136            (DataValue::Time(a), DataValue::Time(b)) => Ok(a.cmp(b)),
137            (DataValue::Timestamp(a), DataValue::Timestamp(b)) => Ok(a.cmp(b)),
138
139            // BIT: lexicographic over the '0'/'1' string representation
140            (DataValue::Bit(a), DataValue::Bit(b)) => Ok(a.cmp(b)),
141
142            // Anything else is a type mismatch
143            _ => Err(ComparisonError::TypeMismatch {
144                left: value_type_name(self).to_string(),
145                right: value_type_name(other).to_string(),
146            }),
147        }
148    }
149}

151// ── NULL-aware wrappers ───────────────────────────────────────────────────────
152
153/// Compare two nullable values following SQL three-valued logic.
154///
155/// Returns:
156/// - `Ok(Some(ordering))` — both operands are non-NULL and comparable.
157/// - `Ok(None)` — at least one operand is `NULL` (result is UNKNOWN).
158/// - `Err(e)` — both operands are non-NULL but incompatible types.
159pub fn compare_nullable(
160    left: Option<&DataValue>,
161    right: Option<&DataValue>,
162) -> Result<Option<Ordering>, ComparisonError> {
163    match (left, right) {
164        (Some(l), Some(r)) => Ok(Some(l.compare(r)?)),
165        // Any NULL operand → UNKNOWN
166        _ => Ok(None),
167    }
168}
169
170/// Test equality of two nullable values following SQL three-valued logic.
171///
172/// Returns `Ok(None)` (UNKNOWN) if either operand is NULL.
173pub fn nullable_equals(
174    left: Option<&DataValue>,
175    right: Option<&DataValue>,
176) -> Result<Option<bool>, ComparisonError> {
177    Ok(compare_nullable(left, right)?.map(|o| o == Ordering::Equal))
178}
